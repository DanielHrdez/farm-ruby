def to_s
    super + "\nType Cattle: #{@type_cattle}\nDestiny: #{@destiny}\nNumber: #{@number}\n" + 
    "Price: #{@price}\nSale Price: #{@sale_price}\nAnimals:\n" + 
    @animals.map { |animal| animal.to_s }.join("\n") + "\n"
end
